make_minimum_required(VERSION 3.10)

# Set your project name
project(BFF)

# C++ standard

    set(CMAKE_BUILD_TYPE /.debug)
endif()


# Include directories
include_directories(${PROJECT_SOURCE_DIR}/src)

# Source files
set(SOURCES
        src/main.cpp

)

# Add executable
add_executable(CimiWeb ${SOURCES}
        /src/main.cpp
)
